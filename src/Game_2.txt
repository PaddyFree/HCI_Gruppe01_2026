import { useEffect, useMemo, useRef, useState } from "react";

const FIELD_SIZES = [9, 16, 25, 36];
const TRIALS_TOTAL = 24;

function getRandomElement(array) {
    return array[Math.floor(Math.random() * array.length)];
}

function shuffleArray(array) {
    const copy = [...array];

    for (let i = copy.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [copy[i], copy[j]] = [copy[j], copy[i]];
    }

    return copy;
}

function createBalancedSoundPlan() {
    const half = TRIALS_TOTAL / 2;

    return shuffleArray([
        ...Array(half).fill("mit_netzbrummen"),
        ...Array(half).fill("ohne_netzbrummen")
    ]);
}

function generateTrial(soundCondition) {
    const size = getRandomElement(FIELD_SIZES);
    const targetIndex = Math.floor(Math.random() * size);

    const items = Array.from({ length: size }, (_, index) => ({
        isTarget: index === targetIndex
    }));

    return {
        size,
        items,
        targetIndex,
        soundCondition
    };
}

function getGridVisuals(size) {
    if (size === 9) {
        return {
            buttonSize: 110,
            dotSize: 28,
            gap: 18
        };
    }

    if (size === 16) {
        return {
            buttonSize: 90,
            dotSize: 24,
            gap: 14
        };
    }

    if (size === 25) {
        return {
            buttonSize: 72,
            dotSize: 20,
            gap: 10
        };
    }

    return {
        buttonSize: 60,
        dotSize: 16,
        gap: 8
    };
}

function formatSoundLabel(soundCondition) {
    return soundCondition === "mit_netzbrummen"
        ? "Mit Netzbrummen"
        : "Ohne Netzbrummen";
}

function getSummary(results) {
    const withHum = results.filter((entry) => entry.soundCondition === "mit_netzbrummen");
    const withoutHum = results.filter((entry) => entry.soundCondition === "ohne_netzbrummen");

    function buildStats(entries) {
        const correctEntries = entries.filter((entry) => entry.correct);
        const errors = entries.filter((entry) => !entry.correct).length;

        const avgTime =
            correctEntries.reduce((sum, entry) => sum + entry.reactionTime, 0) /
            correctEntries.length || 0;

        const accuracy = correctEntries.length / entries.length || 0;

        return {
            total: entries.length,
            errors,
            accuracy,
            avgTime
        };
    }

    return {
        withHum: buildStats(withHum),
        withoutHum: buildStats(withoutHum)
    };
}

export default function App() {
    const [phase, setPhase] = useState("intro");
    const [trialIndex, setTrialIndex] = useState(0);
    const [trial, setTrial] = useState(null);
    const [startTime, setStartTime] = useState(null);
    const [results, setResults] = useState([]);
    const [soundPlan, setSoundPlan] = useState([]);
    const [audioUnlocked, setAudioUnlocked] = useState(false);

    const audioContextRef = useRef(null);
    const humOscillatorRef = useRef(null);
    const humGainRef = useRef(null);

    useEffect(() => {
        return () => {
            stopHum();

            if (audioContextRef.current) {
                audioContextRef.current.close();
                audioContextRef.current = null;
            }
        };
    }, []);

    useEffect(() => {
        if (!trial || phase !== "trial") {
            stopHum();
            return;
        }

        if (trial.soundCondition === "mit_netzbrummen") {
            startHum();
        } else {
            stopHum();
        }

        return () => {
            stopHum();
        };
    }, [trial, phase]);

    const summary = useMemo(() => getSummary(results), [results]);

    async function ensureAudioReady() {
        try {
            if (!audioContextRef.current) {
                const AudioContextClass = window.AudioContext || window.webkitAudioContext;
                audioContextRef.current = new AudioContextClass();
            }

            if (audioContextRef.current.state === "suspended") {
                await audioContextRef.current.resume();
            }

            setAudioUnlocked(true);
            return true;
        } catch (error) {
            console.error("Audio konnte nicht initialisiert werden:", error);
            setAudioUnlocked(false);
            return false;
        }
    }

    function startHum() {
        const audioContext = audioContextRef.current;

        if (!audioContext) {
            return;
        }

        stopHum();

        const fundamental = audioContext.createOscillator();
        const overtone1 = audioContext.createOscillator();
        const overtone2 = audioContext.createOscillator();
        const gain = audioContext.createGain();

        fundamental.type = "sine";
        fundamental.frequency.value = 50;

        overtone1.type = "sine";
        overtone1.frequency.value = 100;

        overtone2.type = "sine";
        overtone2.frequency.value = 150;

        gain.gain.value = 0.08;

        fundamental.connect(gain);
        overtone1.connect(gain);
        overtone2.connect(gain);
        gain.connect(audioContext.destination);

        fundamental.start();
        overtone1.start();
        overtone2.start();

        humOscillatorRef.current = {
            fundamental,
            overtone1,
            overtone2
        };
        humGainRef.current = gain;
    }

    function stopHum() {
        if (humOscillatorRef.current) {
            try {
                humOscillatorRef.current.fundamental.stop();
            } catch {/* all good */ }
            try {
                humOscillatorRef.current.overtone1.stop();
            } catch {/* all good */ }
            try {
                humOscillatorRef.current.overtone2.stop();
            } catch {/* all good */ }
            humOscillatorRef.current.fundamental.disconnect();
            humOscillatorRef.current.overtone1.disconnect();
            humOscillatorRef.current.overtone2.disconnect();
            humOscillatorRef.current = null;
        }

        if (humGainRef.current) {
            humGainRef.current.disconnect();
            humGainRef.current = null;
        }
    }

    function prepareNextTrial(currentTrialIndex, currentSoundPlan) {
        const soundCondition = currentSoundPlan[currentTrialIndex];
        const newTrial = generateTrial(soundCondition);

        setTrial(newTrial);
        setStartTime(Date.now());
        setPhase("trial");
    }

    async function handleStartExperiment() {
        const audioReady = await ensureAudioReady();

        if (!audioReady) {
            alert(
                "Audio konnte im Browser nicht freigeschaltet werden. Ohne Nutzerinteraktion blockieren manche Browser den Ton."
            );
            return;
        }

        const newSoundPlan = createBalancedSoundPlan();
        const firstTrial = generateTrial(newSoundPlan[0]);

        setSoundPlan(newSoundPlan);
        setTrialIndex(0);
        setResults([]);
        setTrial(firstTrial);
        setStartTime(Date.now());
        setPhase("trial");
    }

    function finishTrial(clickedItem, event) {
        if (!trial || startTime === null || phase !== "trial") {
            return;
        }

        stopHum();

        const reactionTime = Date.now() - startTime;
        const correct = clickedItem.isTarget;
        const actualPointerType = event.nativeEvent.pointerType || "unbekannt";

        const nextResults = [
            ...results,
            {
                trialNumber: results.length + 1,
                fieldSize: trial.size,
                soundCondition: trial.soundCondition,
                soundLabel: formatSoundLabel(trial.soundCondition),
                reactionTime,
                correct,
                actualPointerType
            }
        ];

        const nextTrialIndex = trialIndex + 1;

        setResults(nextResults);
        setTrialIndex(nextTrialIndex);

        if (nextTrialIndex >= TRIALS_TOTAL) {
            setTrial(null);
            setStartTime(null);
            setPhase("results");
            return;
        }

        prepareNextTrial(nextTrialIndex, soundPlan);
    }

    if (phase === "intro") {
        return (
            <div style={pageStyle}>
                <div style={cardStyle}>
                    <h1 style={{ marginTop: 0, fontSize: 40 }}>
                        Visuelle Suche mit Netzbrummen
                    </h1>

                    <p>
                        In jedem Trial gibt es genau <b>einen roten Punkt</b> in einem Feld aus grauen
                        Punkten.
                    </p>

                    <p>
                        Deine Aufgabe ist es, den roten Punkt so schnell und so korrekt wie möglich
                        anzuklicken oder anzutippen.
                    </p>

                    <p>
                        Manche Trials laufen intern mit <b>Netzbrummen</b>, andere <b>ohne
                        Netzbrummen</b>.
                    </p>

                    <p>
                        Nach dem Start läuft das Experiment ohne Unterbrechung direkt von Feld zu Feld.
                    </p>

                    <p>
                        Insgesamt gibt es <b>{TRIALS_TOTAL} Trials</b>.
                    </p>

                    <br />

                    <button onClick={handleStartExperiment} style={buttonStyle}>
                        Experiment starten
                    </button>

                    <p style={{ marginTop: 16, color: "#666", fontSize: 14 }}>
                        Audio bereit: {audioUnlocked ? "Ja" : "Nein"}
                    </p>
                </div>
            </div>
        );
    }

    if (phase === "results") {
        return (
            <div style={pageStyle}>
                <div style={{ ...cardStyle, maxWidth: 1100 }}>
                    <h1 style={{ marginTop: 0 }}>Ergebnisse</h1>

                    <div style={summaryGridStyle}>
                        <div style={summaryCardStyle}>
                            <h2 style={{ marginTop: 0, color: "darkgray" }}>Mit Netzbrummen</h2>
                            <p>Trials: {summary.withHum.total}</p>
                            <p>Fehler: {summary.withHum.errors}</p>
                            <p>Trefferquote: {(summary.withHum.accuracy * 100).toFixed(1)}%</p>
                            <p>Ø Reaktionszeit: {Math.round(summary.withHum.avgTime)} ms</p>
                        </div>

                        <div style={summaryCardStyle}>
                            <h2 style={{ marginTop: 0, color: "darkgray" }}>Ohne Netzbrummen</h2>
                            <p>Trials: {summary.withoutHum.total}</p>
                            <p>Fehler: {summary.withoutHum.errors}</p>
                            <p>Trefferquote: {(summary.withoutHum.accuracy * 100).toFixed(1)}%</p>
                            <p>Ø Reaktionszeit: {Math.round(summary.withoutHum.avgTime)} ms</p>
                        </div>
                    </div>

                    <h2>Alle Trials</h2>

                    <div style={tableWrapperStyle}>
                        <table style={tableStyle}>
                            <thead>
                            <tr>
                                <th style={tableHeaderStyle}>Trial</th>
                                <th style={tableHeaderStyle}>Audio</th>
                                <th style={tableHeaderStyle}>Feldgröße</th>
                                <th style={tableHeaderStyle}>PointerType</th>
                                <th style={tableHeaderStyle}>Reaktionszeit</th>
                                <th style={tableHeaderStyle}>Korrekt</th>
                            </tr>
                            </thead>
                            <tbody>
                            {results.map((entry) => (
                                <tr key={entry.trialNumber}>
                                    <td style={tableCellStyle}>{entry.trialNumber}</td>
                                    <td style={tableCellStyle}>{entry.soundLabel}</td>
                                    <td style={tableCellStyle}>{entry.fieldSize}</td>
                                    <td style={tableCellStyle}>{entry.actualPointerType}</td>
                                    <td style={tableCellStyle}>{entry.reactionTime} ms</td>
                                    <td
                                        style={{
                                            ...tableCellStyle,
                                            color: entry.correct ? "green" : "red",
                                            fontWeight: 600
                                        }}
                                    >
                                        {entry.correct ? "Ja" : "Nein"}
                                    </td>
                                </tr>
                            ))}
                            </tbody>
                        </table>
                    </div>
                </div>
            </div>
        );
    }

    if (!trial) {
        return null;
    }

    const visuals = getGridVisuals(trial.size);

    return (
        <div style={pageStyle}>
            <div
                style={{
                    display: "grid",
                    gridTemplateColumns: `repeat(${Math.sqrt(trial.size)}, 1fr)`,
                    gap: visuals.gap,
                    justifyContent: "center"
                }}
            >
                {trial.items.map((item, index) => (
                    <button
                        key={index}
                        onPointerDown={(event) => finishTrial(item, event)}
                        style={{
                            width: visuals.buttonSize,
                            height: visuals.buttonSize,
                            background: "darkgray",
                            display: "flex",
                            alignItems: "center",
                            justifyContent: "center",
                            border: "1px solid #bdbdbd",
                            borderRadius: 8,
                            cursor: "pointer",
                            padding: 0
                        }}
                    >
                        <div
                            style={{
                                width: visuals.dotSize,
                                height: visuals.dotSize,
                                borderRadius: "50%",
                                background: item.isTarget ? "red" : "#666"
                            }}
                        />
                    </button>
                ))}
            </div>
        </div>
    );
}

const pageStyle = {
    minHeight: "100vh",
    display: "flex",
    alignItems: "center",
    justifyContent: "center",
    flexDirection: "column",
    gap: 20,
    padding: 24,
    boxSizing: "border-box",
    background: "#16171C"
};

const cardStyle = {
    width: "100%",
    maxWidth: 760,
    background: "#16171C",
    border: "1px solid #ddd",
    borderRadius: 12,
    padding: 24,
    boxSizing: "border-box",
    boxShadow: "0 8px 24px rgba(0, 0, 0, 0.08)"
};

const buttonStyle = {
    padding: "12px 24px",
    fontSize: 16,
    cursor: "pointer",
    borderRadius: 8,
    border: "1px solid #ccc",
    background: "black",
    color: "white"
};

const summaryGridStyle = {
    display: "grid",
    gridTemplateColumns: "repeat(auto-fit, minmax(280px, 1fr))",
    gap: 16,
    marginBottom: 24
};

const summaryCardStyle = {
    border: "1px solid #ddd",
    borderRadius: 10,
    padding: 16,
    background: "#fafafa"
};

const tableWrapperStyle = {
    width: "100%",
    overflowX: "auto",
    border: "1px solid #ddd",
    borderRadius: 10,
    background: "white"
};

const tableStyle = {
    width: "100%",
    borderCollapse: "collapse"
};

const tableHeaderStyle = {
    borderBottom: "2px solid #ccc",
    padding: "10px 8px",
    textAlign: "left",
    background: "#f0f0f0"
};

const tableCellStyle = {
    borderBottom: "1px solid #e0e0e0",
    padding: "8px"
};
