function getRandomElement(array) {
    return array[Math.floor(Math.random() * array.length)];
}

function shuffleArray(array) {
    const copy = [...array];

    for (let i = copy.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [copy[i], copy[j]] = [copy[j], copy[i]];
    }

    return copy;
}

function getInputStats(results, inputMethod) {
    const methodResults = results.filter(
        (r) => r.assignedInputMethod === inputMethod
    );
    const correctTrials = methodResults.filter((r) => r.correct);
    const errors = methodResults.filter((r) => !r.correct).length;

    const avgTime =
        correctTrials.reduce((sum, r) => sum + r.reactionTime, 0) /
        correctTrials.length || 0;

    const accuracy = correctTrials.length / methodResults.length || 0;

    const difficultyStats = DIFFICULTIES.map((difficulty) => {
        const levelResults = methodResults.filter(
            (r) => r.difficulty === difficulty.level
        );
        const levelCorrect = levelResults.filter((r) => r.correct);
        const levelErrors = levelResults.filter((r) => !r.correct).length;

        const levelAvgTime =
            levelCorrect.reduce((sum, r) => sum + r.reactionTime, 0) /
            levelCorrect.length || 0;

        const levelAccuracy = levelCorrect.length / levelResults.length || 0;

        return {
            level: difficulty.level,
            total: levelResults.length,
            errors: levelErrors,
            accuracy: levelAccuracy,
            avgTime: levelAvgTime
        };
    });

    return {
        inputMethod,
        total: methodResults.length,
        errors,
        accuracy,
        avgTime,
        difficultyStats
    };
}
